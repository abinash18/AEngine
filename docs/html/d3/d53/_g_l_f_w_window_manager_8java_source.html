123 
124  /*
125  * Since using a for loop iterating over the hash map will cause a concurrent
126  * modification exception, we use the iterator which iterates over the entris in
127  * the map.
128  */
129  for (Iterator<Map.Entry<GLFWWindowContext, CopyOnWriteArrayList<GLFWWindow>>> entries = windows.entrySet()
130  .iterator(); entries.hasNext();) {
131  /*
132  * Current Entry, we cannot call next() on the iterator every time we need the
133  * entry because then it will skip to the next entry in line.
134  */
135  Map.Entry<GLFWWindowContext, CopyOnWriteArrayList<GLFWWindow>> entry = entries.next();
136 
137  /* Iterates over the windows which share the context. */
138  for (Iterator<GLFWWindow> subEntrys = entry.getValue().iterator(); subEntrys.hasNext();) {
139  GLFWWindow wnd = subEntrys.next();
140  if (wnd.getGlfw_Handle() == NULL) {
141  continue;
142  }
143 
144  /*
145  * TODO: Add Background Render optimization by breaking out of the loop if the
146  * window is not in focus or iconified.
147  */
148  /*
149  * if (!wnd.isFocused()) { break; }
150  */
151 
152  /*
153  * Sets the current context to the one provided in the entry so we can render
154  * that context.
155  */
156  setContext(wnd.getGlfw_Handle());
157  /*
158  * This will set the capabilities to render the current context on the window.
159  */
160  GL.setCapabilities(wnd.getCapabilities());
161  /* We are ready to render the context now. */
162  wnd.render();
163  }
164 
165  }
166  setContext(NULL);
167  /*
168  * We must set the capabilities to null, because we dont want any operations
169  * done any where else to affect the last window we just rendered.
170  */
171  GL.setCapabilities(null);
172  }

185  public void input(float delta) {
186  /* If there are no contexts left stop the engine. */
187  if (windows.size() == 0) {
188  raiseStopFlag();
189  return;
190  }
191 
192  /* Iterates over everything contained in the hash map to */
193  for (Iterator<Map.Entry<GLFWWindowContext, CopyOnWriteArrayList<GLFWWindow>>> entries = windows.entrySet()
194  .iterator(); entries.hasNext();) {
195 
196  Map.Entry<GLFWWindowContext, CopyOnWriteArrayList<GLFWWindow>> entry = entries.next();
197 
198  for (int i = 0; i < entry.getValue().size(); i++) {
199  GLFWWindow wnd = entry.getValue().get(i);
200 
201  /* Check if the window is destroyed already or not */
202  if (wnd.getGlfw_Handle() != NULL) {
203  if (entry.getKey().context == NULL) {
204  entry.getKey().context = wnd.getGlfw_Handle();
205  }
206  /*
207  * Sets the current context to the one provided in the entry so we can render
208  * that context.
209  */
210  setContext(wnd.getGlfw_Handle());
211  GL.setCapabilities(wnd.getCapabilities());
212  /*
213  * These operations are done in here because I call input first core engine. It
214  * would not make sense and have terrible consequences if it was done in a
215  * method called later.
216  */
217  if (checkClose(wnd)) {
218  /* Dispose the window destroying its context and capability's. */
219  wnd.dispose();
220  logger.debug("Window Destroyed. " + wnd.getWindowName());
221 
222  /*
223  * Break, so we don't cause a null pointer exception if we try updating the
224  * input for a empty window and one we have removed from the array list.
225  */
226  continue;
227  }
228  /* Updates the input for the window. */
229  wnd.input(delta);
230  } else { /* If so then delete the entry. */
231  /* Remove the window from the list. */
232  entry.getValue().remove(i);
233  }
234  }
235  }
236  /*
237  * Resets the context and capabilities to NULL so no action called on the
238  * current thread after the loop will have effect.
239  */
240  setContext(NULL);
241  GL.setCapabilities(null);
242  }
