280  public synchronized <T> void load(String fileName, Class<T> type, AssetLoaderParameters<T> parameter) {
281  AssetLoader loader = getLoader(type, fileName);
282  if (loader == null)
283  throw new AERuntimeException("No loader for type: " + type.getSimpleName());
284 
285  // reset stats
286  if (loadQueue.size() == 0) {
287  loaded = 0;
288  toLoad = 0;
289  peakTasks = 0;
290  }
291 
292  // check if an asset with the same name but a different type has already been
293  // added.
294 
295  // check preload queue
296  for (int i = 0; i < loadQueue.size(); i++) {
297  AssetClassifier desc = loadQueue.get(i);
298  if (desc.getFileName().equals(fileName) && !desc.getType().equals(type))
299  throw new AERuntimeException("AssetI with name '" + fileName
300  + "' already in preload queue, but has different type (expected: " + type.getSimpleName()
301  + ", found: " + desc.getType().getSimpleName() + ")");
302  }
303 
304  // check task list
305  for (int i = 0; i < tasks.size(); i++) {
306  AssetClassifier desc = tasks.get(i).assetDesc;
307  if (desc.getFileName().equals(fileName) && !desc.getType().equals(type))
308  throw new AERuntimeException(
309  "AssetI with name '" + fileName + "' already in task list, but has different type (expected: "
310  + type.getSimpleName() + ", found: " + desc.getType().getSimpleName() + ")");
311  }
312 
313  // check loaded assets
314  if (store.contains(type, fileName)) {
315  throw new AERuntimeException(
316  "AssetI with name '" + fileName + "' already loaded : " + type.getSimpleName());
317  }
318  toLoad++;
319  AssetClassifier assetDesc = new AssetClassifier(fileName, type, parameter);
320  loadQueue.add(assetDesc);
321  logger.debug("Queued: " + assetDesc);
322  }

340  public synchronized boolean update() {
341  try {
342  if (tasks.size() == 0) {
343  // loop until we have a new task ready to be processed
344  while (loadQueue.size() != 0 && tasks.size() == 0) {
345  nextTask();
346  }
347  // have we not found a task? We are done!
348  if (tasks.size() == 0) {
349  executor.dispose();
350  return true;
351  }
352  }
353  return updateTask() && loadQueue.size() == 0 && tasks.size() == 0;
354  } catch (Exception t) {
355  handleTaskError(t);
356  return loadQueue.size() == 0;
357  }

488  private boolean updateTask() {
489  AssetLoadTask task = tasks.peek();
490 
491  boolean complete = true;
492  try {
493  complete = task.cancel || task.update();
494  } catch (RuntimeException ex) {
495  task.cancel = true;
496  taskFailed(task.assetDesc, ex);
497  }
498 
499  // if the task has been cancelled or has finished loading
500  if (complete) {
501  // increase the number of loaded assets and pop the task from the stack
502  if (tasks.size() == 1) {
503  loaded++;
504  peakTasks = 0;
505  }
506  tasks.pop();
507 
508  if (task.cancel)
509  return true;
510 
511  AssetContainer c = addAsset(task.assetDesc.getFileName(), task.assetDesc.getType(), task.getAsset());
512 
513  // otherwise, if a listener was found in the parameter invoke it
514  if (task.assetDesc.getParameter() != null && task.assetDesc.getParameter().loadedCallback != null) {
515  task.assetDesc.getParameter().loadedCallback.finishedLoading(this, task.assetDesc.getFileName(), c);
516  }
517 
518  return true;
519  }
520  return false;
521  }
