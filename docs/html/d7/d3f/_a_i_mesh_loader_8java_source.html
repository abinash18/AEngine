337  AIFileReadProcI fileReadProc = new AIFileReadProc() {
338  public long invoke(long pFile, long pBuffer, long size, long count) {
339  long max = Math.min(data.remaining(), size * count);
340  memCopy(memAddress(data) + data.position(), pBuffer, max);
341  return max;
342  }
343  };
344  AIFileSeekI fileSeekProc = new AIFileSeek() {
345  public int invoke(long pFile, long offset, int origin) {
346  if (origin == Assimp.aiOrigin_CUR) {
347  data.position(data.position() + (int) offset);
348  } else if (origin == Assimp.aiOrigin_SET) {
349  data.position((int) offset);
350  } else if (origin == Assimp.aiOrigin_END) {
351  data.position(data.limit() + (int) offset);
352  }
353  return 0;
354  }
355  };
356  AIFileTellProcI fileTellProc = new AIFileTellProc() {
357  public long invoke(long pFile) {
358  return data.limit();
359  }
360  };
361  aiFile.ReadProc(fileReadProc);
362  aiFile.SeekProc(fileSeekProc);
363  aiFile.FileSizeProc(fileTellProc);
