305  * Calculates the starting position of the uniform variable type and name
306  * definition in the line.
307  */

312

314

325 
326  /*
327  * Finds the names of the uniforms defined inside the struct.
328  */
329  // The position of the first line end character in the struct will mean the end
330  // of the uniform definition in the struct.
331  int componentLineEndCharPos = shaderText.indexOf(LINE_END_CHAR, bracketBegin);
332 
333  /*
334  * Loops while there is a semicolon pos and while it is not greater than that of
335  * the end of the struct end bracket.
336  */
337  while (componentLineEndCharPos != -1 && componentLineEndCharPos < bracketEnd) {
338 
339  int componentNameEnd = componentLineEndCharPos + 1;
340 
341  while (Character.isWhitespace(shaderText.charAt(componentNameEnd - 1))
342  || shaderText.charAt(componentNameEnd - 1) == ';')
343  componentNameEnd--;
344 
345  int componentNameStart = componentLineEndCharPos;
346 
347  /*
348  * While the char it is looping over is not a white space. Because we are going
349  * backwards from the uniform definition. As soon as we hit a space it means we
350  * are at the end of the name of the uniform.
351  */
352  while (!Character.isWhitespace(shaderText.charAt(componentNameStart - 1))) {
353  componentNameStart--; // Back up a character.
354  }
355 
356  int componentTypeEnd = componentNameStart;
357 
358  while (Character.isWhitespace(shaderText.charAt(componentTypeEnd - 1)))
359  componentTypeEnd--;
360 
361  int componentTypeStart = componentTypeEnd;
362 
363  /*
364  * ComponentTypeEnd is one char before the component name start, it is the end
365  * char pos of the type. And this loop will find the position at which it
366  * starts, it loops as long as there is no white space at the character it is
367  * on.
368  */
369  while (!Character.isWhitespace(shaderText.charAt(componentTypeStart - 1))) {
370  componentTypeStart--; // Back up a character.
371  }
372 
373  String componentName = shaderText.substring(componentNameStart, componentLineEndCharPos);
374  String componentType = shaderText.substring(componentTypeStart, componentTypeEnd);
375 
376  // structComponents.add(shaderText.substring(componentNameStart,
377  // componentLineEndCharPos));
378 
379  // System.out.println("'" + shaderText.substring(componentNameStart,
380  // componentLineEndCharPos) + "'");
381 
382  GLSLStruct glslStruct = new GLSLStruct();
383  glslStruct.name = componentName;
384  glslStruct.type = componentType;
385 
386  glslStructs.add(glslStruct);
387 
388  /*
389  * Finds the next end char pos in the file.
390  */
391  componentLineEndCharPos = shaderText.indexOf(LINE_END_CHAR, componentLineEndCharPos + 1);
392 
393  }

396  result.put(structName, glslStructs);

829  public static ShaderSource loadShaderSource(PathHandle pathToShader, ShaderType type) {
830  StringBuilder shaderSource = new StringBuilder();
831  BufferedReader shaderReader = null;
832 
833  final String INCLUDE_DIRECTIVE = "#include", COMMENT_PREFIX = "//", NATIVE_IMPORT_DIRECTIVE = "#import";
834 
835  // glsl include
836 
837  try {
838  shaderReader = new BufferedReader(new FileReader(pathToShader.getFileInstance()));
839  String line;
840 
841  while ((line = shaderReader.readLine()) != null) {
842 
843  line.replaceAll("[\\n\\t ]", "");
844  line = line.trim();
845 
846  if (line.startsWith(COMMENT_PREFIX)) {
847  continue;
848  }
849 
850  /*
851  * Cleans the line of comments and such.
852  */
853 
854  int commentindex;
855 
856  if (line.contains(COMMENT_PREFIX)) {
857  commentindex = line.indexOf(COMMENT_PREFIX);
858  line = line.substring(0, commentindex);
859 
860  }
861 
862  line = line.trim();
863 
864  if (line.isEmpty()) {
865  continue;
866  }
867 
868  if (line.startsWith(NATIVE_IMPORT_DIRECTIVE)) {
869  processNativeImports(line);
870  } else if (line.startsWith(INCLUDE_DIRECTIVE)) {
871 
872  /*
873  * #include 'file' INCLUDE_DIRECTIVE length puts it at the end of the include
874  * and i add 2 to move it past the first quote to the file name. AND THEN IT
875  * ENDS AT THE END OF THE LINE NO WHITE SPACE AT THE END.
876  */
877  /* This appends the loaded shader to the source of the file */
878  /* Loads the shader specified in the #include */
879  PathHandle path = DEFAULT_SHADER_ASSET_DIRECTORY_PATH.resolveChild(
880  "includes/" + line.substring(INCLUDE_DIRECTIVE.length() + 2, line.length() - 1));
881  shaderSource.append(loadShaderSource(path, ShaderType.AE_HEADER).getSource());
882  /* Figures out the name of the file being included. */
883 
884  } else {
885  shaderSource.append(line).append("\n");
886  }
887  }
888  shaderReader.close();
889  } catch (Exception e) {
890  // e.printStackTrace();
891  logger.error("Unable to parse shader " + pathToShader.toString(), e);
892  logger.info("Exiting...");
893  System.exit(1);
894  }
895  return new ShaderSource(pathToShader.getNameWithoutExtension(), shaderSource.toString(), type);
896  }

901  private static void processNativeImports(String line) {
902  // TODO: Process Natives.
903  // In future convert this to a meta file to get generic function file names
904  // which can be included in the shader loading process.
905  // Like #include or #import AE_Functions:LinearizeDepth
906  // Then the loader will look through this meta file and determine the file name
907  // of the function and include only that instead of all functions in the file.
908  // When loading multiple functions from the file it can be defined as so:
909  // #include or #import AE_Functions:{LinearizeDepth, DeLinearizeDepth, ...}
910  // The AE_Functions file would be in a format such as JSON or XML
911 
912  // The convention will be that all files that contain native functions will have
913  // a prefix of AE_
914  // There will be a big file like AE_Functions which a compiler will run over and
915  // find the functions needed for import.
916  // Those functions will have a import notation as well like:
917 
918  // #import AE_Functions:DeLinearize
919  // float LinearizeDepth(float depth) {

921  // return (2.0 * near * far) / (far + near - z * (far - near));
922  // }

924  // And if there is no notation it will mean the function dose not have any
925  // Dependencies.
926 
927  // The notation will be a line above so the compiler will have to send the line
928  // index one above from the function
929  // to the processNativeImports(String shaderText) function.
