247  public static int getUniformByteSize(int uniSize, int uniType, int uniArrayStride, int uniMatStride) {
248  int auxSize = 0;
249  if (uniArrayStride > 0) {
250  auxSize = uniArrayStride * uniSize;
251  } else if (uniMatStride > 0) {
252  switch (uniType) {
253  case GL_FLOAT_MAT2:
254  case GL_FLOAT_MAT2x3:
255  case GL_FLOAT_MAT2x4:
256  case GL_DOUBLE_MAT2:
257  case GL_DOUBLE_MAT2x3:
258  case GL_DOUBLE_MAT2x4:
259  auxSize = 2 * uniMatStride;
260  break;
261  case GL_FLOAT_MAT3:
262  case GL_FLOAT_MAT3x2:
263  case GL_FLOAT_MAT3x4:
264  case GL_DOUBLE_MAT3:
265  case GL_DOUBLE_MAT3x2:
266  case GL_DOUBLE_MAT3x4:
267  auxSize = 3 * uniMatStride;
268  break;
269  case GL_FLOAT_MAT4:
270  case GL_FLOAT_MAT4x2:
271  case GL_FLOAT_MAT4x3:
272  case GL_DOUBLE_MAT4:
273  case GL_DOUBLE_MAT4x2:
274  case GL_DOUBLE_MAT4x3:
275  auxSize = 4 * uniMatStride;
276  break;
277  }
278  } else {
279  auxSize = spGLSLTypeSize.get(uniType);
280  }
281  return auxSize;
282  }
