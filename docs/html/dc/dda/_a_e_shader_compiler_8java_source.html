569  private static Map<String, GLSLStruct> processStructs(String source) {
570  Map<String, GLSLStruct> structs = new HashMap<String, GLSLStruct>();
571  /*
572  * Find the structs using the keyword.
573  */
574  int structStartIndex = source.indexOf(Keywords.STRUCT.word);
575  while (structStartIndex != -1) {
576  /*
577  * Check if the Keyword is a token and not part of a word.
578  */
579  if (!(structStartIndex != 0
580  && (Character.isWhitespace(source.charAt(structStartIndex - 1))
581  || Operators.SEMICOLON.op.equals(Character.toString(source.charAt(structStartIndex - 1))))
582  && Character.isWhitespace(source.charAt(structStartIndex + Keywords.STRUCT.word.length())))) {
583  structStartIndex = source.indexOf(Keywords.STRUCT.word,
584  structStartIndex + Keywords.STRUCT.word.length());
585  continue;
586  }
587 
588  int beginStructNameDefinitionIndex = structStartIndex + Keywords.STRUCT.word.length() + 1;
589  int structBracketBeginIndex = source.indexOf(Operators.CURLEY_BRACKET_BEGIN.op,
590  beginStructNameDefinitionIndex);
591  int structBracketEndIndex = source.indexOf(Operators.CURLEY_BRACKET_END.op, structBracketBeginIndex);
592 
593  String structName = source.substring(beginStructNameDefinitionIndex, structBracketBeginIndex).trim();
594  // Semi colon at the end of the struct component.
595  int structComponentEndIndex = source.indexOf(Operators.SEMICOLON.op, structBracketBeginIndex);
596  GLSLStruct result = new GLSLStruct();
597  result.components = new ArrayList<GLSLUniform>();
598  while (structComponentEndIndex != -1 && structComponentEndIndex < structBracketEndIndex) {
599  int structComponentNameEnd = structComponentEndIndex + 1;
600  while (Character.isWhitespace(source.charAt(structComponentNameEnd - 1)) || Operators.SEMICOLON.op
601  .equals(Character.toString(source.charAt(structComponentNameEnd - 1)))) {
602  structComponentNameEnd--;
603  }
604 
605  int structComponentNameStart = structComponentEndIndex;
606  while (!Character.isWhitespace(source.charAt(structComponentNameStart - 1))) {
607  structComponentNameStart--; // Back up a character.
608  }
609 
610  int structComponentTypeEnd = structComponentNameStart;
611  while (Character.isWhitespace(source.charAt(structComponentTypeEnd - 1))) {
612  structComponentTypeEnd--;
613  }
614 
615  int structComponentTypeStart = structComponentTypeEnd;
616  while (!Character.isWhitespace(source.charAt(structComponentTypeStart - 1))) {
617  structComponentTypeStart--; // Back up a character.
618  }
619 
620  String componentName = source.substring(structComponentNameStart, structComponentEndIndex);
621  String componentType = source.substring(structComponentTypeStart, structComponentTypeEnd);
622 
623  GLSLUniform structComponent = new GLSLUniform();
624  structComponent.name = componentName;
625  structComponent.type = componentType;
626  result.components.add(structComponent);
627  structComponentEndIndex = source.indexOf(Operators.SEMICOLON.op, structComponentEndIndex + 1);
628  }
629  result.name = structName;
630  structs.put(structName, result);
631  structStartIndex = source.indexOf(Keywords.STRUCT.word, structStartIndex + Keywords.STRUCT.word
632  .length()); /* This is to prevent it from looping over the same uniform keyword. */
633  }
634  return structs;
635  }

783  private static String processImportLine(String line, String callingProgram) {
784  StringBuilder loadedImport = new StringBuilder();
785  /*
786  * Now we need to separate out the import file and programs.
787  */
788  String importFile; // The file from which to import
789  /*
790  * I do this since the user can accidentally have more than one whitespace. So
791  * we find the first one and then token-ize the line using this.
792  */
793  String importDefinition; // There are only two possible tokens the import directive and the import call.
794  String temp = ""; // This is to spell out the token and add it to the array.
795  ArrayList<String> program_names = new ArrayList<String>();
796  /*
797  * First we need to check if the import call is valid.
798  */
799  if (!line.startsWith(Tokens.AE_IMPORT_DIRECTIVE)) {
800  return "";
801  }
802  if (Character.isWhitespace(line.charAt(Tokens.AE_IMPORT_DIRECTIVE.length()))) {
803  int importFileSeperatorIndex = 0;
804  temp = line.substring(Tokens.AE_IMPORT_DIRECTIVE.length(), line.length());
805  temp = temp.replaceAll("\\s+", ""); // Clear all the white space.
806  /*
807  * We substring the import line from the start of the temp to the : char. And
808  * while we do so we also define the index of it to use to separate the program
809  * tokens.
810  */
811  importFile = temp.substring(0, (importFileSeperatorIndex = temp.indexOf(Operators.COLON.op)));
812  /*
813  * Now we separate the dictionary containing the import programs from the line
814  * it self.
815  */
816  importDefinition = temp.substring(importFileSeperatorIndex + 1, temp.length()).replaceAll("\\{", "")
817  .replaceAll("\\}", "");
818  logger.fine("Importing Programs : " + line);
819  String[] programs = importDefinition.split(",");
820  // System.out.println(importDefinition);
821  for (int i = 0; i < programs.length; i++) {
822  programs[i] = programs[i].replaceAll("\"", "").replaceAll("\\s+", "").replaceAll(";", "");
823  if (programs[i].isEmpty())
824  continue;
825  program_names.add(programs[i]);
826  }
827  } else {
828  throw new AERuntimeException(
829  "Invalid Import call in shader line : (Space between #import and file needed, well its not really but i just dont want to write the code to do so. "
830  + line);
831  }
832  for (String program : program_names) {
833  /*
834  * Prevents Recursive overflows, meaning that the same import can't call it self
835  * ever.
836  */
837  if (program.equals(callingProgram)) {
838  throw new AEShaderCompilerRuntimeException(
839  "Recursive Overflow Detected, This import is calling it self. Program: " + callingProgram);
840  }
841  loadedImport.append(processImport(program, importFile));
842  }
843  return loadedImport.toString();
844  }

846  private static String processImport(String programName, String fileName) {
847  StringBuilder program = new StringBuilder();
848  AEShaderGLSLProgram gp = null;
849  /*
850  * Check if the import was in this file or was loaded before.
851  */
852  if ((gp = loadedImports.get(fileName + Tokens.AE_SHADER_FILE_TYPE, programName)) == null) {
853  /*
854  * If it wasn't try finding it then loading it.
855  */
856  loadAllImportsInFile(AEShaderParserYAML
857  .parse(DEFAULT_SHADER_ASSET_DIRECTORY_PATH.resolveChild(fileName + Tokens.AE_SHADER_FILE_TYPE)));
858  /*
859  * If we still can't find it then give up and go throw a exception.
860  */
861  if ((gp = loadedImports.get(fileName + Tokens.AE_SHADER_FILE_TYPE, programName)) == null) {
862  throw new AEShaderCompilerRuntimeException("Import in file not fo)und : " + fileName
863  + Tokens.AE_SHADER_FILE_TYPE + " Import: " + programName);
864  }
865  }
866  /*
867  * Now that we have determined the file exists we can continue. Or we have not
868  * and the program exits. Now we have to find that program.
869  */
870  /*
871  * If the program doesn't exist we throw an exception.
872  */
873  if ((gp = loadedImports.get(fileName + Tokens.AE_SHADER_FILE_TYPE, programName)) == null) {
874  throw new AEShaderCompilerRuntimeException("Shader Import not found in file: " + fileName
875  + Tokens.AE_SHADER_FILE_TYPE + " Import: " + programName);
876  }
877  /*
878  * Otherwise we recursively process it and append it to the original source.
879  */
880  program.append(processShaderSource(gp.getAE_SHADER_GLSL_PROGRAM_SOURCE(), gp.getAE_SHADER_GLSL_PROGRAM_NAME()));
881  return program.toString();
882  }
